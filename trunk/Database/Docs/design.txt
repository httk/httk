Property:
   // General property data common for all properties
   PropertyId              : Integer Key
   ComputationId           : Integer
   PropertyDataTable       : String

PropertyData_FormationEnergy_0K:
   // Property data specific to one type of property
   PropertyId              : Integer
   FormationEnergy         : Float

PropertyData_DOS_0K:
   // Another example of a property data; DOS is big, so it is not in
   // the database. Instead this refers back to computational data.
   PropertyId: Integer
   DataPath                : String // internal path to DOS data
                                    // in the hdf data tree of the
                                    // associated Computation
Geometry:
   // Each row is generated from source crystal structures +
   // a best effort process to generate a unique representation
   // of the crystal structure independent of species assignment
   // and volume. This is what is meant by a 'Geometry'.
   //
   GeometryId              : Integer Key
   UniqueRepresentation    : String
   < further columns describing an abstract crystal structure >

SpeciesMap:
   SpeciesMapId            : Integer Key
   UniqueRepresentation    : String
   // {SpeciesAssignment}  : one -> many table

SpeciesAssignment:
   SpeciesMapId            : Integer
   Site                    : Integer // corresponding to the GeometryId
   Species                 : Integer // atomic number

Computation:
    ComputationId      : Integer Key
    ComputationDate    : Date
    AddedDate          : Date
    GeometryId         : Integer
    SpeciesMapId       : Integer
    Volume             : Float
    CodeId             : Integer
    ProjectId          : Integer
    Hash               : String // Hash value of manifest file that lists hashes of all relevant data of the computation
    // {Related},      : one->many table
    // {Tag}           : one->many table
    // {Owner}         : one->many table

ComputationOwner:
    // Keeps track for every computation
    // who "owns" it; a list of public keys
    ComputationId      : Integer
    String             : Signature // Signs the Hash in the Computation table
    KeyId              : Integer

ComputationRelated:
    ComputationId            : Integer
    RelatedToComputationId   : Integer
    SequenceNumber           : Integer // sets the sequence of related
                                       // runs
    Relation                 : String  // e.g. "Relaxation step"

ComputationTag:
    ComputationId : Integer
    String        : Tag name // e.g., "spin-polarized"
    String        : Tag value // e.g., "true"

Code:
    CodeId              : Integer Key
    Name                : String
    Version             : String (e.g. "2.4.16.32-beta")
    Reference           : String (e.g. a link to the VASP website)

Project:
    ProjectId           : Integer Key
    Description         : String
    // {Owner}          : one->many table

ProjectOwner:
    // Keeps track for every computation
    // who "owns" it; a list of public keys
    ProjectId           : Integer
    KeyId               : Integer

Key:
    // Table to keep track of generated ssl keys to
    // make it possible to handle identification of
    // "owners" of calculations
    KeyId               : Integer Key
    KeyData             : String
