[docs]def atomic_symbol(parse):
    """
    Helper function to produce an atomic symbol if you have some kind of identifier, but does not know what it is.
    """
    try:
        return parse.to_atomic_symbol()
    except AttributeError:
        pass
    if isinstance(parse,int):
        return symbols[parse]
    if isinstance(parse,(list,tuple)):
        return atomic_symbol(parse[0])
    s = str(parse)
    if s in symbols:
        return s
    if s in names:
        return symbols[names.index(s)]
    if s == 'D':
        return 'D'
    raise Exception("Do not know how to parse input: "+str(parse))
