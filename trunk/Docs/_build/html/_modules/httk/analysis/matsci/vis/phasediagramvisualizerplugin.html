[docs]    def show(self,params={},backends=['matplotlib'],debug=False):
        for backend in backends:
            if backend == 'matplotlib':
                try:
                    from matplotlibphasediagramvisualizer import MatplotlibPhaseDiagramVisualizer
                    self.visualizer = MatplotlibPhaseDiagramVisualizer(self.phasediagram)
                    self.visualizer.show()
                    return
                except ImportError:
                    if debug:
                        raise
                    pass
        raise Exception("PhaseDiagramVisualizerPlugin.show: None of the requested / available backends available, tried:"+str(backends))
